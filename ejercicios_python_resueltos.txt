# Pedir cantidad de estudiantes
n = int(input("¿Cuántos estudiantes hay en el taller? "))

edades = []

# Pedir edades una por una
for i in range(n):
    edad = int(input(f"Edad del estudiante {i+1}: "))
    edades.append(edad)

# Imprimir todas las edades
print("\nEdades ingresadas:")
for e in edades:
    print(e)

# Calcular promedio
promedio = sum(edades) / len(edades)

print(f"\nEl promedio de edad del grupo es: {promedio:.2f}")
