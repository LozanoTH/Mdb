# Ejercicio 7 - Circunferencia y diámetro
import math
radio = float(input("Radio del círculo: "))
circunferencia = 2 * math.pi * radio
diametro = 2 * radio
print("Circunferencia:", circunferencia)
print("Diámetro:", diametro)

# Ejercicio 8 - Peso molecular
peso1 = float(input("Peso del primer átomo: "))
atomo1 = int(input("Número de átomos del primero: "))
peso2 = float(input("Peso del segundo átomo: "))
atomo2 = int(input("Número de átomos del segundo: "))
pesoMolecular = peso1 * atomo1 + peso2 * atomo2
print("Peso molecular:", pesoMolecular)

# Ejercicio 9 - Factorial
n = int(input("Número: "))
factorial = 1
for i in range(1, n + 1):
    factorial *= i
print("Factorial:", factorial)

# Ejercicio 10 - CTR (Click Through Rate)
impresiones = int(input("Impresiones: "))
clicks = int(input("Clicks: "))
CTR = (clicks / impresiones) * 100
print("CTR:", CTR, "%")
